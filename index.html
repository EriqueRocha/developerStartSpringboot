terminal — manual installation (recommended for inspection)
step by step

# 1) Install GPG key
curl -fsSL https://eriquerocha.github.io/developerStartSpringboot/public.key | sudo gpg --dearmor -o /usr/share/keyrings/developerstartspringboot-archive-keyring.gpg

# 2) Add repository (amd64)
echo "deb [arch=amd64 signed-by=/usr/share/keyrings/developerstartspringboot-archive-keyring.gpg] https://eriquerocha.github.io/developerStartSpringboot stable main" | sudo tee /etc/apt/sources.list.d/developerstartspringboot.list

# 3) Update and install
sudo apt update
sudo apt install developerstartspringboot

Note: Your system may alert you if you change the Origin/Label/Codename on the first update — this is expected.
